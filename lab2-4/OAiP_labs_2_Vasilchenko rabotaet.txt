#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <conio.h>
#define ARRAY_SIZE 100 
int main()
{
	int i, N,  count = 0;
	printf("enter the number of elements in the array \n");
	scanf("%d", &N);
	int *a = new int[N];

	if (N > 100 || N < 1)
	{
		printf("invalid input");
	}

	else
	{
		for (i = 0; i < N; i++)
		{
			printf("A[%d] = ", i);
			scanf("%d", &a[i]);
		}

		for (i = 0; i < N; i++)
			printf("%d", a[i]);

		for (i = 0; i < N; i++)
			if (a[i] > 0)
			{
				count = count + a[i];
			}

		printf(" sum of positive numbers %d \n", count);
	}
	_getch();
	return 0;
}
